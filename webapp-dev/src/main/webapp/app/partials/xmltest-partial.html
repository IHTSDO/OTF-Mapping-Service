Code: Annotating the Map 

@XmlRootElement(name = "mapTest")
public class MapXmlTestJpa implements MapXmlTest {

	Map map = new HashMap();
	
	public MapXmlTestJpa() {
		
		map.put("key1", "value1");
		map.put("key2", "value2");
		map.put("key3", "value3");
	}

	@XmlJavaTypeAdapter(MapAdapter.class)
	@XmlElement(name = "map")
	public Map getMap() {
		return map;
	}

	public void setMap(Map map) {
		this.map = map;
	}
}
